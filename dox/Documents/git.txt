git config --global user.name "parth patel" # to set name globally

git config credential.helper store # stores password next time you input in plain text

git config user.email parth@patel.com # to set email for current project
git config credential.username #

git status # harmless
git log # harmless
git log --oneline # precise list of commits

git commit --amend # to add remaining file to last commit or correct last commit message
